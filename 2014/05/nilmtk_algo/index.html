Adding a new disaggregation algorithm to NILMTK

Tue 27 May 2014 · tagged nilmtk

In this post I will illustrate how to add a new disaggregation algorithm to NILMTK v0.1. This algorithm is purely for demonstration purpose and is expected to perform really bad! NILMTK is evolving and we might soon hit v0.2 which should make dealing with larger datasets much easier.

Creating a new disaggregation algorithm¶

Every disaggregation algorithm should inherit the abstract Disaggregator class. It must define the following four functions:

  1. Train: Train to create a model
  2. Disaggregate: Disaggregate on test data to create predictions
  3. Export model: Export learnt model to a JSON file
  4. Import model: Import model from JSON file
In [5]:
class RandomPowerDisaggregator(Disaggregator):
    
    def __init__(self):
        self.model = {}
        self.predictions = pd.DataFrame()
        
    def train(self, building, aggregate='mains', submetered='appliances',
              disagg_features=[Measurement('power', 'active')],
              environmental=None):
        """
        This method accepts as input a nilmtk building object, contains
        information on how to do disaggregation: mains->appliances or 
        mains->circuits, features to use for disaggregation
        
        In this particular demo algorithm, the model contains mean power of 
        each appliance
        """
        
        train_appliances = building.utility.electric.get_dataframe_of_appliances(
            measurement=disagg_features[0])

        train_mains = building.utility.electric.get_dataframe_of_mains(
            measurement=disagg_features[0])
        
        for appliance_name, appliance_df in train_appliances.iteritems():
            self.model[appliance_name] = appliance_df.mean()
    
    def disaggregate(self, building, disagg_features=[Measurement('power', 'active')], environmental=None):
        """
        Disaggregate method takes a test building and produces a predicted power time series. It must use
        the same disaggregation features as the training algortihm
        
        In this demo algorithm, each appliance at each time instant is predicted to consume
        its mean power. Not too smart.. But this is only for demo
        """
        
        test_mains = building.utility.electric.get_dataframe_of_mains(measurement=disagg_features[0])
        predicted_power = {}
        for appliance, appliance_mean in self.model.iteritems():
            predicted_power[appliance] = appliance_mean*np.ones(len(test_mains.index))
        
        self.predictions = pd.DataFrame(
            predicted_power, index=test_mains.index)
    
    def export_model(self, filename):
        """
        Exporting learnt model to JSON file
        """
        model_copy = {}
        for appliance, appliance_states in self.model.iteritems():
            model_copy[
                "{}_{}".format(appliance.name, appliance.instance)] = appliance_states
        j = json.dumps(model_copy)
        with open(filename, 'w+') as f:
            f.write(j)
    
    def import_model(self, filename):
        """
        Importing a model from JSON file
        """
        with open(filename, 'r') as f:
            temp = json.loads(f.read())
        for appliance, power in temp.iteritems():
            appliance_name = appliance.split("_")[0].encode("ascii")
            appliance_instance = int(appliance.split("_")[1])
            appliance_name_instance = ApplianceName(
                appliance_name, appliance_instance)
            self.model[appliance_name_instance] = power
